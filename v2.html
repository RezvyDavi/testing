// ==UserScript== // @name Custom User-Agent Override // @namespace http://tampermonkey.net/ // @version 1.0 // @description Override navigator.userAgent string // @author Rezvy // @match *://*/* // @grant none // ==/UserScript== (function () { 'use strict'; // Ubah string ini sesuai user agent yang diinginkan const customUA = "Mozilla/5.0 (Windows NT 10.0; Win64; x64) MyCustomBrowser/1.0"; // Redefine navigator.userAgent Object.defineProperty(navigator, 'userAgent', { value: customUA, configurable: false, writable: false }); // Bonus: juga override navigator.platform jika perlu Object.defineProperty(navigator, 'platform', { value: "Win32", configurable: false, writable: false }); console.log("User-Agent diubah jadi:", navigator.userAgent); })();
